CREATE TABLE IF NOT EXISTS deals (
	id INT AUTO_INCREMENT PRIMARY KEY,
    	name VARCHAR(155) CHARACTER SET utf8,
    	start VARCHAR(25) CHARACTER SET utf8,
    	end VARCHAR(25) CHARACTER SET utf8,
    	image VARCHAR(110) CHARACTER SET utf8,
    	vendor VARCHAR(41) CHARACTER SET utf8,
    	terms VARCHAR(4315) CHARACTER SET utf8,
    	category VARCHAR(126) CHARACTER SET utf8
);
